üêõ Fixed Date Conversion Error

Error: TypeError: value.toISOString is not a function

Solution: Added safe date conversion helper that checks if value is a Date object before calling toISOString().

const safeToISOString = (value: any): string | null => {
  if (value instanceof Date) {
    return value.toISOString();
  }
  if (typeof value === 'string') {
    return value;
  }
  return null;
};
